bootcmd=run boot_a
boot_a=setenv rootpart 2; run boot_common
boot_b=setenv rootpart 3; run boot_common
boot_common=setenv bootargs "${bootargs} root=/dev/mmcblk0p${rootpart} rootwait rw"; \
           fatload mmc 0:1 ${kernel_addr_r} ${kernel_file}; \
           fatload mmc 0:1 ${fdt_addr_r} ${fdt_file}; \
           booti ${kernel_addr_r} - ${fdt_addr_r}
kernel_file=Image
fdt_file=bcm2712-rpi-5-b.dtb
